35      @Test
36      public void 画像を生成() throws Exception {
37          // モックを設定
38          String url = "http://www.plantuml.com/plantuml/uml/SyfFKj2rKt3CoKnELR1Io4ZDoSa70000";
39          byte[] expectedContent = IOUtils.toByteArray(this.getClass().getResourceAsStream("/service/PlantUmlServiceTest/simple_uml.png"));
40          given(this.service.generate(url, FileFormat.PNG)).willReturn(expectedContent);
41  
42          // テストを実行
43          ResultActions result = this.mvc.perform(get("/images?url=http%3a%2f%2fwww%2eplantuml%2ecom%2fplantuml%2fuml%2fSyfFKj2rKt3CoKnELR1Io4ZDoSa70000"));
44  
45          // テスト結果を確認
46          result.andExpect(status().isOk())
47              .andExpect(header().string("X-Api-Version", this.appVersion))
48              .andExpect(header().string("Content-Type", "image/png"))
49              .andExpect(content().bytes(expectedContent));
50      }
