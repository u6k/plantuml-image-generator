3   
4   import java.io.IOException;
5   import java.net.URLDecoder;
6   
7   import io.swagger.annotations.Api;
8   import io.swagger.annotations.ApiOperation;
9   import io.swagger.annotations.ApiParam;
10  import io.swagger.annotations.ApiResponse;
11  import io.swagger.annotations.ApiResponses;
12  import io.swagger.annotations.ResponseHeader;
13  import me.u6k.plantuml_image_generator.service.PlantUmlService;
14  import net.sourceforge.plantuml.FileFormat;
15  import org.apache.commons.lang3.StringUtils;
16  import org.slf4j.Logger;
17  import org.slf4j.LoggerFactory;
18  import org.springframework.beans.factory.annotation.Autowired;
19  import org.springframework.beans.factory.annotation.Value;
20  import org.springframework.http.HttpHeaders;
21  import org.springframework.http.HttpStatus;
22  import org.springframework.http.ResponseEntity;
23  import org.springframework.web.bind.annotation.RequestMapping;
24  import org.springframework.web.bind.annotation.RequestMethod;
25  import org.springframework.web.bind.annotation.RequestParam;
26  import org.springframework.web.bind.annotation.RestController;
27  
28  @RestController
29  @Api
30  public class ImageRestController {
31  
32      private static final Logger L = LoggerFactory.getLogger(ImageRestController.class);
33  
34      @Autowired
35      private PlantUmlService plantUmlService;
36  
37      @Value("${info.app.version}")
38      private String appVersion;
39  
40      @ApiOperation(value = "Generate PlantUML Image(png or svg)", protocols = "https", produces = "image/png,image/svg+xml", responseHeaders = { @ResponseHeader(name = "X-Api-Version", description = "APIのバージョン番号") })
41      @ApiResponses(value = {
42          @ApiResponse(code = 400, message = "パラメータが不正な場合"),
43          @ApiResponse(code = 500, message = "PlantUML文書の取得に失敗した場合")
44      })
45      @RequestMapping(value = "/images", method = RequestMethod.GET)
46      public ResponseEntity<byte[]> generateImage(
47          @RequestParam(name = "url", required = true) //
48          @ApiParam(name = "url", required = true, value = "PlantUML文書が存在するURL。URLエンコードしてください。", example = "http%3A%2F%2Fwww.plantuml.com%2Fplantuml%2Fuml%2FSyfFKj2rKt3CoKnELR1Io4ZDoSa70000") //
49          String encodedUrl,
50          @RequestParam(name = "format", required = false) //
51          @ApiParam(name = "format", required = false, value = "画像形式。\"png\"または\"svg\"。", defaultValue = "png", example = "svg") //
52          String format) throws IOException {
53          L.debug("#generateImage: encodedUrl={}, format={}", encodedUrl, format);
54  
55          /*
56           * 入力チェック
57           */
58          if (StringUtils.isBlank(encodedUrl)) {
59              throw new IllegalArgumentException("encodedUrl is blank.");
60          }
61  
62          FileFormat fileFormat;
63          if (StringUtils.isBlank(format)) {
64              fileFormat = FileFormat.PNG;
65          } else if (StringUtils.equalsIgnoreCase(format, "png")) {
66              fileFormat = FileFormat.PNG;
67          } else if (StringUtils.equalsIgnoreCase(format, "svg")) {
68              fileFormat = FileFormat.SVG;
69          } else {
70              throw new IllegalArgumentException("\"" + format + "\" is unknown format.");
71          }
72  
73          /*
74           * URLから画像データを作成
75           */
76          String url = URLDecoder.decode(encodedUrl, "UTF-8");
77          L.debug("url decoded: {}", url);
78  
79          byte[] image = this.plantUmlService.generate(url, fileFormat);
80  
81          /*
82           * レスポンスを構築
83           */
84          L.debug("build response data start:");
85          HttpHeaders headers = new HttpHeaders();
86          switch (fileFormat) {
87              case PNG:
88                  headers.add("Content-Type", "image/png");
89                  break;
90              case SVG:
91                  headers.add("Content-Type", "image/svg+xml");
92                  break;
93              default:
94                  throw new IllegalArgumentException("\"" + format + "\" is unknown format.");
95          }
96          headers.add("X-Api-Version", this.appVersion);
97  
98          ResponseEntity<byte[]> resp = new ResponseEntity<byte[]>(image, headers, HttpStatus.OK);
99          L.debug("build response data end: length={}", image.length);
100 
101         return resp;
102     }
