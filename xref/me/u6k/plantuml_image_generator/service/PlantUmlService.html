2   package me.u6k.plantuml_image_generator.service;
3   
4   import java.io.ByteArrayOutputStream;
5   import java.io.IOException;
6   import java.net.URLDecoder;
7   
8   import me.u6k.plantuml_image_generator.model.WebRepository;
9   import net.sourceforge.plantuml.SourceStringReader;
10  import org.apache.commons.lang3.StringUtils;
11  import org.slf4j.Logger;
12  import org.slf4j.LoggerFactory;
13  import org.springframework.beans.factory.annotation.Autowired;
14  import org.springframework.stereotype.Service;
15  
16  @Service
17  public class PlantUmlService {
18  
19      private static final Logger L = LoggerFactory.getLogger(PlantUmlService.class);
20  
21      @Autowired
22      private WebRepository webRepo;
23  
24      public byte[] generate(String url) throws IOException {
25          /*
26           * 入力チェック
27           */
28          L.debug("#generate: url={}", url);
29          if (StringUtils.isBlank(url)) {
30              throw new IllegalArgumentException("url is blank.");
31          }
32  
33          /*
34           * URLからデータを取得
35           */
36          url = URLDecoder.decode(url, "UTF-8");
37          L.debug("url decoded: url={}", url);
38  
39          String umlText = this.webRepo.getUrl(url);
40  
41          /*
42           * データからPlantUML画像を生成
43           */
44          byte[] image;
45  
46          try (ByteArrayOutputStream out = new ByteArrayOutputStream()) {
47              L.debug("plantuml image generate start:");
48              SourceStringReader reader = new SourceStringReader(umlText);
49              reader.generateImage(out); // FIXME: 戻り値を処理
50              L.debug("plantuml image generate end:");
51  
52              image = out.toByteArray();
53              L.debug("image.length={}", image.length);
54          }
55  
56          return image;
57      }
58  
59  }
