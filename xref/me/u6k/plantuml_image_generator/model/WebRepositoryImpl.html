2   package me.u6k.plantuml_image_generator.model;
3   
4   import java.io.IOException;
5   
6   import me.u6k.plantuml_image_generator.exception.ReadWebResourceException;
7   import org.apache.commons.lang3.StringUtils;
8   import org.apache.http.client.config.RequestConfig;
9   import org.apache.http.client.methods.CloseableHttpResponse;
10  import org.apache.http.client.methods.HttpGet;
11  import org.apache.http.impl.client.CloseableHttpClient;
12  import org.apache.http.impl.client.HttpClients;
13  import org.apache.http.util.EntityUtils;
14  import org.slf4j.Logger;
15  import org.slf4j.LoggerFactory;
16  import org.springframework.beans.factory.annotation.Value;
17  import org.springframework.http.HttpStatus;
18  import org.springframework.stereotype.Repository;
19  
20  @Repository
21  public class WebRepositoryImpl implements WebRepository {
22  
23      private static final int TIMEOUT_LENGTH = 10000;
24  
25      private static final Logger L = LoggerFactory.getLogger(WebRepositoryImpl.class);
26  
27      @Value("${info.app.groupId}")
28      private String appGroupId;
29  
30      @Value("${info.app.name}")
31      private String appName;
32  
33      @Value("${info.app.version}")
34      private String appVersion;
35  
36      @Override
37      public String getUrl(String url) {
38          L.debug("#getUrl: url={}", url);
39  
40          if (StringUtils.isBlank(url)) {
41              throw new IllegalArgumentException("url is blank.");
42          }
43  
44          try (CloseableHttpClient httpClient = HttpClients.createDefault()) {
45              HttpGet httpGet = new HttpGet(url);
46  
47              RequestConfig requestConfig = RequestConfig.custom()
48                  .setConnectTimeout(TIMEOUT_LENGTH)
49                  .setSocketTimeout(TIMEOUT_LENGTH)
50                  .build();
51              httpGet.setConfig(requestConfig);
52              L.debug("set config: timeout={}", TIMEOUT_LENGTH);
53  
54              String userAgent = this.appGroupId + ":" + this.appName + "/" + this.appVersion;
55              httpGet.addHeader("User-Agent", userAgent);
56              L.debug("set config: user-agent={}", userAgent);
57  
58              // TODO: キャッシュを活用
59  
60              L.debug("request: url={}", url);
61              try (CloseableHttpResponse httpResp = httpClient.execute(httpGet)) {
62                  L.debug("response: status={}", httpResp.getStatusLine().getStatusCode());
63                  if (httpResp.getStatusLine().getStatusCode() != HttpStatus.OK.value()) {
64                      throw new ReadWebResourceException("url=" + url + ", statusCode=" + httpResp.getStatusLine().getStatusCode());
65                  }
66  
67                  String responseText = EntityUtils.toString(httpResp.getEntity());
68                  L.debug("response: text.length={}", responseText.length());
69  
70                  return responseText;
71              }
72          } catch (IOException e) {
73              throw new ReadWebResourceException(e);
74          }
75      }
76  
77  }
